Option / 參數

	$.jkrolling.apply( $('#demo'), './demo.jpg', {
		sliceNum : 300, // 切片數量 | set slice num (for more smooth)
		centerKeepRange : 0.2, // 中間保留範圍
		rollingDeep : 0.5, // 滾動深度
		rollingType : $.jkrolling.type_straight // 滾動類型
	});
